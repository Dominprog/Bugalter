Структура данных
Описание таблиц и полей базы данных

Таблица employees

Поле Тип Описание
id INTEGER Уникальный идентификатор
full_name VARCHAR(200) ФИО сотрудника
hire_date DATE Дата приема на работу
position VARCHAR(100) Должность
salary_type VARCHAR(20) Тип оплаты (оклад/почасовая)
salary DECIMAL(10,2) Оклад
department_id INTEGER ID отдела
status VARCHAR(20) Статус сотрудника

Таблица departments

Поле Тип Описание
id INTEGER Уникальный идентификатор
name VARCHAR(100) Название отдела
budget DECIMAL(10,2) Бюджет отдела
Примеры использования
Практические примеры работы с API

Получение сотрудников с отделами

async function loadEmployeesWithDepartments() {
  try {
    const [employees, departments] = await Promise.all([
      fetch('http://localhost:3000/employees').then(r => r.json()),
      fetch('http://localhost:3000/departments').then(r => r.json())
    ]);
    
    const employeesWithDept = employees.map(emp => ({
      ...emp,
      department: departments.find(dept => dept.id === emp.department_id)
    }));
    
    return employeesWithDept;
  } catch (error) {
    console.error('Ошибка загрузки:', error);
  }
}

Расчет фонда оплаты труда

async function calculateSalaryFund() {
  const employees = await fetch('http://localhost:3000/employees')
    .then(response => response.json());
  
  const totalSalary = employees.reduce((sum, emp) => {
    return sum + parseFloat(emp.salary || 0);
  }, 0);
  
  return totalSalary;
}
